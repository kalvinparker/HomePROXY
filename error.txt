#
# Site Definitions
# We will get certificates for the DuckDNS domain and Caddy will
# automatically serve them for your No-IP domain as well.
#

# Proxy Host for Vaultwarden
83-validator.duckdns.org {
    # This handle block matches requests for your No-IP domain
    # and proxies them to Vaultwarden.
    @vault host vault.83.ddns.net
    handle @vault {
        reverse_proxy 192.168.1.6:8080
    }

    # This handle block matches requests for your DuckDNS domain
    # and proxies them to Vaultwarden as well.
    handle {
        reverse_proxy 192.168.1.6:8080
    }

    # Tell Caddy HOW to get a certificate for this site
    tls {
        dns duckdns {$DUCKDNS_TOKEN}
    }
}
